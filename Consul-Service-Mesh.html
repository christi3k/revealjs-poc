Code Sample

              
              service {
                name = "redis"
                port = 8000
                tags = ["global"]

                checks = [
                  {
                    id = "redis_network"
                    tcp = "localhost:8000"
                    timeout = "1s"
                  },
                  {
                    id = "redis_script"
                    script = "/usr/local/bin/check_redis.py"
                    interval = "10s"
                  }
                ]
              }
